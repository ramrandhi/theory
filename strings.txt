Example:

public class StringMethodsExample {
    public static void main(String[] args) {
        String str = "Hello, World!";

        // Character extraction
        System.out.println(str.charAt(0));  // Output: H

        // Substring
        System.out.println(str.substring(7));  // Output: World!

        // Case conversion
        System.out.println(str.toLowerCase());  // Output: hello, world!

        // Search
        System.out.println(str.indexOf("World"));  // Output: 7

        // Replacement
        System.out.println(str.replace("World", "Java"));  // Output: Hello, Java!

        // Splitting
        String[] parts = str.split(", ");
        System.out.println(parts[0]);  // Output: Hello

        // Trimming
        System.out.println("  Hello  ".trim());  // Output: Hello
    }
}
